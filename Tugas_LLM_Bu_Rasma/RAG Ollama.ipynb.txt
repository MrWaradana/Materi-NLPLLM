    "\n",
    "def build_prompt(user_q, retrieved):\n",
    "    # build a context string listing retrieved items with small excerpts\n",
    "    context_blocks = []\n",
    "    for i, r in enumerate(retrieved, 1):\n",
    "        # limit length to avoid too long prompt\n",
    "        excerpt = shorten(r[\"text\"], width=600, placeholder=\"...\")\n",
    "        context_blocks.append(f\"[{i}] {excerpt}\")\n",
    "    context = \"\\n\\n\".join(context_blocks)\n",
    "    prompt = (\n",
    "        SYSTEM_INSTR + \"\\n\\n\"\n",
    "        f\"Context (reference QA pairs):\\n{context}\\n\\n\"\n",
    "        f\"Pertanyaan pengguna: {user_q}\\n\\n\"\n",
    "        \"Gunakan konteks di atas untuk menyusun jawaban baru (jangan menambahkan fakta yang tidak ada). \"\n",
    "        \"Cantumkan juga sumber dalam format: Sumber: [index].\"\n",
    "    )\n",
    "    return prompt\n",
